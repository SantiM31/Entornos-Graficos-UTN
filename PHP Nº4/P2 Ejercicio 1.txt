<?php
$a = array( 'color' => 'rojo',
   'sabor' => 'dulce',
   'forma' => 'redonda',
   'nombre' => 'manzana',
   4
   );
?>


<?php
$a['color'] = 'rojo';
$a['sabor'] = 'dulce';
$a['forma'] = 'redonda';
$a['nombre'] = 'manzana';
$a[] = 4;
?>


  - Son equivalentes. La manera de crear el array es distinta pero el resultado es el mismo.
